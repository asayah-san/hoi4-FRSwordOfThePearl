    id = siam.12
    title = siam.12.t
    desc = siam.12.d
    picture = GFX_report_event_dutch_soldiers_indonesia

    mean_time_to_happen = {
        days = 1
    }

    trigger = {
        original_tag = SIA
        if = {
            limit = { has_war_with = ENG }
            controls_state = 1007
        }

        if = {
            limit = {
                has_war_with = FRA
                has_war_with = FIC
            }
            OR = {
                AND = {
                    controls_state = 962
                    controls_state = 963
                    controls_state = 741

                    controls_state = 959
                    controls_state = 960
                    controls_state = 670
                }
                FIC = { has_capitulated = yes }
            }
        }
    }

    option = {
        name = siam.12.a

        if = {
            limit = { has_war_with = ENG }
            every_country = {
                limit = { is_in_faction_with = ENG }
                whitepeace = SIA
            }
        }

        if = {
            limit = { has_war_with = FRA }
            every_country = {
                limit = { is_in_faction_with = FRA }
                whitepeace = FRA
            }
        }
    }
}
